217  void
218  rowSlice2DSplit(Series& series, MeshRecordComponent& rho)
219  {
220  Extent meshExtent = rho.getExtent();
221 
222  if ((unsigned int)m_MPISize > meshExtent[1])
223  return;
224 
225  benchmark::Timer rowTime( "Row slice time, divide among all ranks", m_MPIRank, program_start );
226  auto blob = meshExtent[1] / m_MPISize;
227 
228  // not going through all rows. up to first <m_MPISize> rows
229  for (unsigned int row=0; row < meshExtent[0]; row++)
230  {
231  if (row >= (unsigned int) m_MPISize) break;
232 
233  Offset rowOff = {row, m_MPIRank * blob};
234  Extent rowExt = {1, blob};
235  if (row == (meshExtent[0] - 1))
236  rowExt[1] = meshExtent[1] - rowOff[1];
237  auto row_data = rho.loadChunk<double>(rowOff, rowExt);
238  series.flush();
239  }
240  }

249  void
250  colSlice2DSplit( Series& series, MeshRecordComponent& rho )
251  {
252  Extent meshExtent = rho.getExtent();
253 
254  if ( (unsigned int)m_MPISize > meshExtent[0] )
255  return;
256 
257  if ( meshExtent[0] % m_MPISize != 0 )
258  return;
259 
260  benchmark::Timer colTime( "Col slice time, divided load", m_MPIRank, program_start );
261  auto blob = meshExtent[0] / m_MPISize;
262 
263  for (unsigned int col = 0; col < meshExtent[1]; col++)
264  {
265  if ( col >= (unsigned int) m_MPISize ) break;
266 
267  Offset colOff = {m_MPIRank*blob, col};
268  Extent colExt = {blob, 1};
269  auto col_data = rho.loadChunk<double>(colOff, colExt);
270  series.flush();
271  }

282  void
283  readStep(Series& series, int ts)
284  {
285  std::string comp_name = openPMD::MeshRecordComponent::SCALAR;
286  MeshRecordComponent rho = series.iterations[ts].meshes["rho"][comp_name];
287  Extent meshExtent = rho.getExtent();
288 
289  if ( 2 == meshExtent.size() )
290  {
291  if ( 0 == m_MPIRank )
292  std::cout << "... rho meshExtent : ts=" << ts << " ["
293  << meshExtent[0] << "," << meshExtent[1] << "]" << std::endl;
294 
295  if ( m_Pattern % 3 == 0) {
296  rowSlice2D(series, rho, false);
297  colSlice2D(series, rho, false);
298  }
299 
300  if ( m_Pattern % 2 == 0 ) {
301  rowSlice2DSplit(series, rho);
302  colSlice2DSplit(series, rho);
303  }
304 
305  if ( m_Pattern % 5 == 0 ) {
306  rowSlice2D(series, rho, true);
307  colSlice2D(series, rho, true);
308  }
309  }
310 
311  // reading particles
312  if ( m_Pattern % 7 == 0 )
313  {
314  openPMD::ParticleSpecies electrons =
315  series.iterations[ts].particles["ion"];
316  RecordComponent charge = electrons["charge"][RecordComponent::SCALAR];
317  sliceParticles(series, charge);
318  }
319  }

328  void sliceParticles(Series& series, RecordComponent& charge)
329  {
330  Extent pExtent = charge.getExtent();
331 
332  auto blob = pExtent[0]/(10*m_MPISize);
333  if (0 == blob)
334  return;
335 
336  auto start = pExtent[0]/4;
337 
338  std::ostringstream s;
339  s << "particle retrievel time, ["<<start<<" + "<< (blob * m_MPISize) <<"] ";
340 
341  benchmark::Timer colTime( s.str().c_str(), m_MPIRank, program_start );
342 
343  Offset colOff = {m_MPIRank*blob};
344  Extent colExt = {blob};
345  auto col_data = charge.loadChunk<double>(colOff, colExt);
346  series.flush();
347  }
348 
349  int m_MPISize = 1;
350  int m_MPIRank = 0;
351  int m_Pattern = 30;
352  std::string m_Backend = ".bp";
353 };
