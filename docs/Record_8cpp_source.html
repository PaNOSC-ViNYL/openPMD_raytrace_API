48 void
49 Record::flush_impl(std::string const& name)
50 {
51  if(IOHandler->m_frontendAccess == Access::READ_ONLY )
52  {
53  for( auto& comp : *this )
54  comp.second.flush(comp.first);
55  } else
56  {
57  if( !written() )
58  {
59  if( scalar() )
60  {
61  RecordComponent& rc = at(RecordComponent::SCALAR);
62  rc.m_writable->parent = parent;
63  rc.parent = parent;
64  rc.flush(name);
65  IOHandler->flush();
66  m_writable->abstractFilePosition = rc.m_writable->abstractFilePosition;
67  rc.abstractFilePosition = m_writable->abstractFilePosition.get();
68  abstractFilePosition = rc.abstractFilePosition;
69  written() = true;
70  } else
71  {
72  Parameter< Operation::CREATE_PATH > pCreate;
73  pCreate.path = name;
74  IOHandler->enqueue(IOTask(this, pCreate));
75  for( auto& comp : *this )
76  comp.second.parent = getWritable(this);
77  }
78  }
79 
80  for( auto& comp : *this )
81  comp.second.flush(comp.first);
82 
83  flushAttributes();
84  }
85 }

87 void
88 Record::read()
89 {
90  if( scalar() )
91  {
92  /* using operator[] will incorrectly update parent */
93  this->at(RecordComponent::SCALAR).read();
94  } else
95  {
96  written() = false;
97  clear_unchecked();
98  written() = true;
99  Parameter< Operation::LIST_PATHS > pList;
100  IOHandler->enqueue(IOTask(this, pList));
101  IOHandler->flush();
102 
103  Parameter< Operation::OPEN_PATH > pOpen;
104  for( auto const& component : *pList.paths )
105  {
106  RecordComponent& rc = (*this)[component];
107  pOpen.path = component;
108  IOHandler->enqueue(IOTask(&rc, pOpen));
109  *rc.m_isConstant = true;
110  rc.read();
111  }
112 
113  Parameter< Operation::LIST_DATASETS > dList;
114  IOHandler->enqueue(IOTask(this, dList));
115  IOHandler->flush();
116 
117  Parameter< Operation::OPEN_DATASET > dOpen;
118  for( auto const& component : *dList.datasets )
119  {
120  RecordComponent& rc = (*this)[component];
121  dOpen.name = component;
122  IOHandler->enqueue(IOTask(&rc, dOpen));
123  IOHandler->flush();
124  rc.written() = false;
125  rc.resetDataset(Dataset(*dOpen.dtype, *dOpen.extent));
126  rc.written() = true;
127  rc.read();
128  }
129  }
130 
131  readBase();
132 
133  readAttributes();
134 }
