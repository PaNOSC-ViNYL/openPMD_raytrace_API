30  Format
31  determineFormat(std::string const &filename) {
32  if (auxiliary::ends_with(filename, ".h5"))
33  return Format::HDF5;
34  if (auxiliary::ends_with(filename, ".bp")) {
35  auto const bp_backend = auxiliary::getEnvString(
36  "OPENPMD_BP_BACKEND",
37 #if openPMD_HAVE_ADIOS2
38  "ADIOS2"
39 #elif openPMD_HAVE_ADIOS1
40  "ADIOS1"
41 #else
42  "ADIOS2"
43 #endif
44  );
45 
46  if (bp_backend == "ADIOS2")
47  return Format::ADIOS2;
48  else if (bp_backend == "ADIOS1")
49  return Format::ADIOS1;
50  else
51  throw std::runtime_error(
52  "Environment variable OPENPMD_BP_BACKEND for .bp backend is neither ADIOS1 nor ADIOS2: " +
53  bp_backend
54  );
55  }
56  if (auxiliary::ends_with(filename, ".json"))
57  return Format::JSON;
58  if (std::string::npos != filename.find('.') /* extension is provided */ )
59  throw std::runtime_error("Unknown file format. Did you append a valid filename extension?");
60 
61  return Format::DUMMY;
62  }

69  case Format::ADIOS1:
70  case Format::ADIOS2:

72  case Format::JSON:
73  return ".json";
74  default:
75  return "";
76  }
