134 void
135 RecordComponent::flush(std::string const& name)
136 {
137  if(IOHandler->m_frontendAccess == Access::READ_ONLY )
138  {
139  while( !m_chunks->empty() )
140  {
141  IOHandler->enqueue(m_chunks->front());
142  m_chunks->pop();
143  }
144  } else
145  {
146  if( !written() )
147  {
148  if( constant() )
149  {
150  Parameter< Operation::CREATE_PATH > pCreate;
151  pCreate.path = name;
152  IOHandler->enqueue(IOTask(this, pCreate));
153  Parameter< Operation::WRITE_ATT > aWrite;
154  aWrite.name = "value";
155  aWrite.dtype = m_constantValue->dtype;
156  aWrite.resource = m_constantValue->getResource();
157  IOHandler->enqueue(IOTask(this, aWrite));
158  aWrite.name = "shape";
159  Attribute a(getExtent());
160  aWrite.dtype = a.dtype;
161  aWrite.resource = a.getResource();
162  IOHandler->enqueue(IOTask(this, aWrite));
163  } else
164  {
165  Parameter< Operation::CREATE_DATASET > dCreate;
166  dCreate.name = name;
167  dCreate.extent = getExtent();
168  dCreate.dtype = getDatatype();
169  dCreate.chunkSize = m_dataset->chunkSize;
170  dCreate.compression = m_dataset->compression;
171  dCreate.transform = m_dataset->transform;
172  IOHandler->enqueue(IOTask(this, dCreate));
173  }
174  }
175 
176  while( !m_chunks->empty() )
177  {
178  IOHandler->enqueue(m_chunks->front());
179  m_chunks->pop();
180  }
181 
182  flushAttributes();
183  }

192 void
193 RecordComponent::readBase()
194 {
195  using DT = Datatype;
196  Parameter< Operation::READ_ATT > aRead;
197 
198  if( constant() )
199  {
200  aRead.name = "value";
201  IOHandler->enqueue(IOTask(this, aRead));
202  IOHandler->flush();
203 
204  Attribute a(*aRead.resource);
205  DT dtype = *aRead.dtype;
206  written() = false;
207  switch( dtype )
208  {
209  case DT::LONG_DOUBLE:
210  makeConstant(a.get< long double >());
211  break;
212  case DT::DOUBLE:
213  makeConstant(a.get< double >());
214  break;
215  case DT::FLOAT:
216  makeConstant(a.get< float >());
217  break;
218  case DT::CLONG_DOUBLE:
219  makeConstant(a.get< std::complex< long double > >());
220  break;
221  case DT::CDOUBLE:
222  makeConstant(a.get< std::complex< double > >());
223  break;
224  case DT::CFLOAT:
225  makeConstant(a.get< std::complex< float > >());
226  break;
227  case DT::SHORT:
228  makeConstant(a.get< short >());
229  break;
230  case DT::INT:
231  makeConstant(a.get< int >());
232  break;
233  case DT::LONG:
234  makeConstant(a.get< long >());
235  break;
236  case DT::LONGLONG:
237  makeConstant(a.get< long long >());
238  break;
239  case DT::USHORT:
240  makeConstant(a.get< unsigned short >());
241  break;
242  case DT::UINT:
243  makeConstant(a.get< unsigned int >());
244  break;
245  case DT::ULONG:
246  makeConstant(a.get< unsigned long >());
247  break;
248  case DT::ULONGLONG:
249  makeConstant(a.get< unsigned long long >());
250  break;
251  case DT::CHAR:
252  makeConstant(a.get< char >());
253  break;
254  case DT::UCHAR:
255  makeConstant(a.get< unsigned char >());
256  break;
257  case DT::BOOL:
258  makeConstant(a.get< bool >());
259  break;
260  default:
261  throw std::runtime_error("Unexpected constant datatype");
262  }
263  written() = true;
264 
265  aRead.name = "shape";
266  IOHandler->enqueue(IOTask(this, aRead));
267  IOHandler->flush();
268  a = Attribute(*aRead.resource);
269  Extent e;
270 
271  // uint64_t check
272  Datatype const attrDtype = *aRead.dtype;
273  if( isSame( attrDtype, determineDatatype< uint64_t >() ) )
274  e.push_back( a.get< uint64_t >() );
275  else if( isSame( attrDtype, determineDatatype< std::vector< uint64_t > >() ) )
276  for( auto const& val : a.get< std::vector< uint64_t > >() )
277  e.push_back( val );
278  else
279  {
280  std::ostringstream oss;
281  oss << "Unexpected datatype ("
282  << *aRead.dtype
283  << ") for attribute 'shape' ("
284  << determineDatatype< uint64_t >()
285  << " aka uint64_t)";
286  throw std::runtime_error(oss.str());
287  }
288 
289  written() = false;
290  resetDataset(Dataset(dtype, e));
291  written() = true;
292  }
293 
294  aRead.name = "unitSI";
295  IOHandler->enqueue(IOTask(this, aRead));
296  IOHandler->flush();
297  if( *aRead.dtype == DT::DOUBLE )
298  setUnitSI(Attribute(*aRead.resource).get< double >());
299  else
300  throw std::runtime_error("Unexpected Attribute datatype for 'unitSI'");
301 
302  readAttributes();
303 }
