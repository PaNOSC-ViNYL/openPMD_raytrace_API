246 template< typename T >
247 inline std::shared_ptr< T >
248 RecordComponent::loadChunk(Offset o, Extent e, double targetUnitSI)
249 {
250  uint8_t dim = getDimensionality();
251 
252  // default arguments
253  // offset = {0u}: expand to right dim {0u, 0u, ...}
254  Offset offset = o;
255  if( o.size() == 1u && o.at(0) == 0u && dim > 1u )
256  offset = Offset(dim, 0u);
257 
258  // extent = {-1u}: take full size
259  Extent extent(dim, 1u);
260  if( e.size() == 1u && e.at(0) == -1u )
261  {
262  extent = getExtent();
263  for( uint8_t i = 0u; i < dim; ++i )
264  extent[i] -= offset[i];
265  }
266  else
267  extent = e;
268 
269  uint64_t numPoints = 1u;
270  for( auto const& dimensionSize : extent )
271  numPoints *= dimensionSize;
272 
273  auto newData = std::shared_ptr<T>(new T[numPoints], []( T *p ){ delete [] p; });
274  loadChunk(newData, offset, extent, targetUnitSI);
275  return newData;
276 }

278 template< typename T >
279 inline void
280 RecordComponent::loadChunk(std::shared_ptr< T > data, Offset o, Extent e, double targetUnitSI)
281 {
282  if( !std::isnan(targetUnitSI) )
283  throw std::runtime_error("unitSI scaling during chunk loading not yet implemented");
284  Datatype dtype = determineDatatype(data);
285  if( dtype != getDatatype() )
286  if( !isSameInteger< T >( dtype ) &&
287  !isSameFloatingPoint< T >( dtype ) &&
288  !isSameComplexFloatingPoint< T >( dtype ) )
289  throw std::runtime_error("Type conversion during chunk loading not yet implemented");
290 
291  uint8_t dim = getDimensionality();
292 
293  // default arguments
294  // offset = {0u}: expand to right dim {0u, 0u, ...}
295  Offset offset = o;
296  if( o.size() == 1u && o.at(0) == 0u && dim > 1u )
297  offset = Offset(dim, 0u);
298 
299  // extent = {-1u}: take full size
300  Extent extent(dim, 1u);
301  if( e.size() == 1u && e.at(0) == -1u )
302  {
303  extent = getExtent();
304  for( uint8_t i = 0u; i < dim; ++i )
305  extent[i] -= offset[i];
306  }
307  else
308  extent = e;
309 
310  if( extent.size() != dim || offset.size() != dim )
311  {
312  std::ostringstream oss;
313  oss << "Dimensionality of chunk ("
314  << "offset=" << offset.size() << "D, "
315  << "extent=" << extent.size() << "D) "
316  << "and record component ("
317  << int(dim) << "D) "
318  << "do not match.";
319  throw std::runtime_error(oss.str());
320  }
321  Extent dse = getExtent();
322  for( uint8_t i = 0; i < dim; ++i )
323  if( dse[i] < offset[i] + extent[i] )
324  throw std::runtime_error("Chunk does not reside inside dataset (Dimension on index " + std::to_string(i)
325  + " - DS: " + std::to_string(dse[i])
326  + " - Chunk: " + std::to_string(offset[i] + extent[i])
327  + ")");
328  if( !data )
329  throw std::runtime_error("Unallocated pointer passed during chunk loading.");
330 
331  if( constant() )
332  {
333  uint64_t numPoints = 1u;
334  for( auto const& dimensionSize : extent )
335  numPoints *= dimensionSize;
336 
337  T value = m_constantValue->get< T >();
338 
339  T* raw_ptr = data.get();
340  std::fill(raw_ptr, raw_ptr + numPoints, value);
341  } else
342  {
343  Parameter< Operation::READ_DATASET > dRead;
344  dRead.offset = offset;
345  dRead.extent = extent;
346  dRead.dtype = getDatatype();
347  dRead.data = std::static_pointer_cast< void >(data);
348  m_chunks->push(IOTask(this, dRead));
349  }
350 }

352 template< typename T >
353 inline void
354 RecordComponent::storeChunk(std::shared_ptr<T> data, Offset o, Extent e)
355 {
356  if( constant() )
357  throw std::runtime_error("Chunks cannot be written for a constant RecordComponent.");
358  if( empty() )
359  throw std::runtime_error("Chunks cannot be written for an empty RecordComponent.");
360  if( !data )
361  throw std::runtime_error("Unallocated pointer passed during chunk store.");
362  Datatype dtype = determineDatatype(data);
363  if( dtype != getDatatype() )
364  {
365  std::ostringstream oss;
366  oss << "Datatypes of chunk data ("
367  << dtype
368  << ") and record component ("
369  << getDatatype()
370  << ") do not match.";
371  throw std::runtime_error(oss.str());
372  }
373  uint8_t dim = getDimensionality();
374  if( e.size() != dim || o.size() != dim )
375  {
376  std::ostringstream oss;
377  oss << "Dimensionality of chunk ("
378  << "offset=" << o.size() << "D, "
379  << "extent=" << e.size() << "D) "
380  << "and record component ("
381  << int(dim) << "D) "
382  << "do not match.";
383  throw std::runtime_error(oss.str());
384  }
385  Extent dse = getExtent();
386  for( uint8_t i = 0; i < dim; ++i )
387  if( dse[i] < o[i] + e[i] )
388  throw std::runtime_error("Chunk does not reside inside dataset (Dimension on index " + std::to_string(i)
389  + " - DS: " + std::to_string(dse[i])
390  + " - Chunk: " + std::to_string(o[i] + e[i])
391  + ")");
392 
393  Parameter< Operation::WRITE_DATASET > dWrite;
394  dWrite.offset = o;
395  dWrite.extent = e;
396  dWrite.dtype = dtype;
397  /* std::static_pointer_cast correctly reference-counts the pointer */
398  dWrite.data = std::static_pointer_cast< void const >(data);
399  m_chunks->push(IOTask(this, dWrite));
400 }
