49 createData( const unsigned long& size, const T& val, bool increment = false )
50 {
51  auto E = std::shared_ptr< T > {
52  new T[size], []( T * d ){ delete[] d; }

54 
55  for( unsigned long i = 0ul; i < size; i++ )
56  {
57  if(increment)

116  void GetRankCountOffset(unsigned long& offset, unsigned long& count, int& step) const
117  {
118  count = m_Bulk;
119  offset = m_Bulk * m_MPIRank;
120 
121  if (!m_Unbalance)
122  return;
123 
124  if (m_MPISize < 2)
125  return;
126 
127  if (step % 3 != 1)
128  return;
129 
130  if (m_MPIRank % 10 == 0)
131  count = 0;
132 
133  if (m_MPIRank % 10 == 1)
134  {
135  offset -= m_Bulk;
136  count += m_Bulk;

145  void setBlockDistributionInRank(int step)
146  {
147  unsigned long rankOffset, rankCount;
148  GetRankCountOffset(rankOffset, rankCount, step);
149 
150  if (0 == rankCount)
151  return;
152 
153  // many small writes
154  srand(time(NULL) * (m_MPIRank + m_MPISize) );
155 
156  auto nBlocks = GetSeg();
157  if ((rankCount / nBlocks) <= 1)
158  nBlocks = 1;
159 
160  m_InRankDistribution.clear();
161  unsigned long counter = 0ul;
162  for( unsigned long i = 0ul; i < nBlocks; i++ )
163  {
164  unsigned long blockSize = rankCount/nBlocks;
165  if ((rankCount % nBlocks != 0) && (i == (nBlocks -1)))
166  blockSize = rankCount - blockSize * (nBlocks -1);
167 
168  m_InRankDistribution.push_back(std::make_pair(rankOffset + counter, blockSize));
169  counter += blockSize;
170  }
171  }

235  {

372  unsigned long
373  getNthMeshExtent( unsigned int n, Offset& offset, Extent& count )
374  {
375  if (n >= getNumBlocks())
376  return 0;
377 
378  if (1 == m_GlobalMesh.size())
379  {
380  offset = {m_InRankDistribution[n].first};
381  count = {m_InRankDistribution[n].second};
382  return count[0];
383  }
384 
385  if (2 == m_GlobalMesh.size())
386  {
387  auto mid = m_GlobalMesh[1]/2;
388  auto rest = m_GlobalMesh[1] - mid;
389  auto ss = m_InRankDistribution.size();
390  if (n < ss)
391  {
392  offset = {m_InRankDistribution[n].first, 0};
393  count = {m_InRankDistribution[n].second, mid};
394  }
395  else
396  { // ss <= n << 2*ss
397  offset = {m_InRankDistribution[n-ss].first, rest};
398  count = {m_InRankDistribution[n-ss].second, rest};
399  }
400 
401  return count[0] * count[1];
402  }
403 
404  return 0;
405  }

426  void
427  getNthParticleExtent( unsigned int n, unsigned long& offset, unsigned long& count )
428  {
429  if ( n >= getNumBlocks() )
430  return;
431 
432  if ( 1 == m_GlobalMesh.size() )
433  {
434  offset = m_InRankDistribution[n].first * m_Ratio ;
435  count = m_InRankDistribution[n].second * m_Ratio ;
436  return;
437  }
438 
439  if ( 2 == m_GlobalMesh.size() )
440  {
441  auto mid = m_GlobalMesh[1]/2;
442  auto rest = m_GlobalMesh[1] - mid;
443  auto ss = m_InRankDistribution.size();
444 
445  auto rankPatch = m_Bulk * mid * m_MPIRank * m_Ratio;
446  if ( n < ss)
447  {
448  offset = rankPatch + m_InRankDistribution[n].first * mid * m_Ratio;
449  count = m_InRankDistribution[n].second * mid * m_Ratio;
450  }
451  else
452  {
453  auto firstHalf = m_Bulk * mid * m_Ratio + rankPatch;
454  offset = m_InRankDistribution[n - ss].first * rest * m_Ratio + firstHalf;
455  count = m_InRankDistribution[n - ss].second * rest * m_Ratio;
456  }
457  }
458  }
