107 Iteration &
108 Iteration::close( bool _flush )
109 {
110  using bool_type = unsigned char;
111  if( this->IOHandler->m_frontendAccess != Access::READ_ONLY )
112  {
113  setAttribute< bool_type >( "closed", 1u );
114  }
115  *m_closed = CloseStatus::ClosedInFrontend;
116  if( _flush )
117  {
118  /* Find the root point [Series] of this file,
119  * meshesPath and particlesPath are stored there */
120  Writable *w = this->parent;
121  while( w->parent )
122  w = w->parent;
123 
124  auto s = dynamic_cast< Series* >( w->attributable );
125  if( s == nullptr )
126  throw std::runtime_error("[Iteration::close] Series* is a nullptr");
127 
128  // figure out my iteration number
129  uint64_t index;
130  bool found = false;
131  for( auto const & pair : s->iterations )
132  {
133  if( pair.second.m_writable.get() == this->m_writable.get() )
134  {
135  found = true;
136  index = pair.first;
137  break;
138  }
139  }
140  if( !found )
141  {
142  throw std::runtime_error(
143  "[Iteration::close] Iteration not found in Series." );
144  }
145  std::map< uint64_t, Iteration > flushOnlyThisIteration{
146  { index, *this } };
147  switch( *s->m_iterationEncoding )
148  {
149  using IE = IterationEncoding;
150  case IE::fileBased:
151  s->flushFileBased( flushOnlyThisIteration );
152  break;
153  case IE::groupBased:
154  s->flushGroupBased( flushOnlyThisIteration );
155  break;
156  }
157  }
158  return *this;
159 }

181 void
182 Iteration::flushFileBased(std::string const& filename, uint64_t i)
183 {
184  /* Find the root point [Series] of this file,
185  * meshesPath and particlesPath are stored there */
186  Writable *w = this->parent;
187  while( w->parent )
188  w = w->parent;
189 
190  auto s = dynamic_cast< Series* >( w->attributable );
191  if( s == nullptr )
192  throw std::runtime_error("[Iteration::flushFileBased] Series* is a nullptr");
193 
194  if( !written() )
195  {
196  /* create file */
197  Parameter< Operation::CREATE_FILE > fCreate;
198  fCreate.name = filename;
199  IOHandler->enqueue(IOTask(s, fCreate));
200 
201  /* create basePath */
202  Parameter< Operation::CREATE_PATH > pCreate;
203  pCreate.path = auxiliary::replace_first(s->basePath(), "%T/", "");
204  IOHandler->enqueue(IOTask(&s->iterations, pCreate));
205 
206  /* create iteration path */
207  pCreate.path = std::to_string(i);
208  IOHandler->enqueue(IOTask(this, pCreate));
209  } else
210  {
211  // operations for create mode
212  if((IOHandler->m_frontendAccess == Access::CREATE ) &&
213  ( (IOHandler->backendName() == "MPI_ADIOS1") || (IOHandler->backendName() == "ADIOS1") ) )
214  {
215  Parameter< Operation::OPEN_FILE > fOpen;
216  fOpen.name = filename;
217  IOHandler->enqueue(IOTask(s, fOpen));
218  flush();
219 
220  return;
221  }
222 
223  // operations for read/read-write mode
224  /* open file */
225  Parameter< Operation::OPEN_FILE > fOpen;
226  fOpen.name = filename;
227  IOHandler->enqueue(IOTask(s, fOpen));
228 
229  /* open basePath */
230  Parameter< Operation::OPEN_PATH > pOpen;
231  pOpen.path = auxiliary::replace_first(s->basePath(), "%T/", "");
232  IOHandler->enqueue(IOTask(&s->iterations, pOpen));
233 
234  /* open iteration path */
235  pOpen.path = std::to_string(i);
236  IOHandler->enqueue(IOTask(this, pOpen));
237  }
238 
239  flush();
240 }

256 void
257 Iteration::flush()
258 {
259  if(IOHandler->m_frontendAccess == Access::READ_ONLY )
260  {
261  for( auto& m : meshes )
262  m.second.flush(m.first);
263  for( auto& species : particles )
264  species.second.flush(species.first);
265  } else
266  {
267  /* Find the root point [Series] of this file,
268  * meshesPath and particlesPath are stored there */
269  Writable *w = this->parent;
270  while( w->parent )
271  w = w->parent;
272  auto s = dynamic_cast< Series* >(w->attributable);
273 
274  if( !meshes.empty() || s->containsAttribute("meshesPath") )
275  {
276  if( !s->containsAttribute("meshesPath") )
277  s->setMeshesPath("meshes/");
278  s->flushMeshesPath();
279  meshes.flush(s->meshesPath());
280  for( auto& m : meshes )
281  m.second.flush(m.first);
282  }
283 
284  if( !particles.empty() || s->containsAttribute("particlesPath") )
285  {
286  if( !s->containsAttribute("particlesPath") )
287  s->setParticlesPath("particles/");
288  s->flushParticlesPath();
289  particles.flush(s->particlesPath());
290  for( auto& species : particles )
291  species.second.flush(species.first);
292  }
293 
294  flushAttributes();
295  }

298 void
299 Iteration::read()
300 {
301  using DT = Datatype;
302  Parameter< Operation::READ_ATT > aRead;
303 
304  aRead.name = "dt";
305  IOHandler->enqueue(IOTask(this, aRead));
306  IOHandler->flush();
307  if( *aRead.dtype == DT::FLOAT )
308  setDt(Attribute(*aRead.resource).get< float >());
309  else if( *aRead.dtype == DT::DOUBLE )
310  setDt(Attribute(*aRead.resource).get< double >());
311  else
312  throw std::runtime_error("Unexpected Attribute datatype for 'dt'");
313 
314  aRead.name = "time";
315  IOHandler->enqueue(IOTask(this, aRead));
316  IOHandler->flush();
317  if( *aRead.dtype == DT::FLOAT )
318  setTime(Attribute(*aRead.resource).get< float >());
319  else if( *aRead.dtype == DT::DOUBLE )
320  setTime(Attribute(*aRead.resource).get< double >());
321  else
322  throw std::runtime_error("Unexpected Attribute datatype for 'time'");
323 
324  aRead.name = "timeUnitSI";
325  IOHandler->enqueue(IOTask(this, aRead));
326  IOHandler->flush();
327  if( *aRead.dtype == DT::DOUBLE )
328  setTimeUnitSI(Attribute(*aRead.resource).get< double >());
329  else
330  throw std::runtime_error("Unexpected Attribute datatype for 'timeUnitSI'");
331 
332  /* Find the root point [Series] of this file,
333  * meshesPath and particlesPath are stored there */
334  Writable *w = getWritable(this);
335  while( w->parent )
336  w = w->parent;
337  auto s = dynamic_cast< Series* >(w->attributable);
338 
339  Parameter< Operation::LIST_PATHS > pList;
340  std::string version = s->openPMD();
341  bool hasMeshes = false;
342  bool hasParticles = false;
343  if( version == "1.0.0" || version == "1.0.1" )
344  {
345  IOHandler->enqueue(IOTask(this, pList));
346  IOHandler->flush();
347  hasMeshes = std::count(
348  pList.paths->begin(),
349  pList.paths->end(),
350  auxiliary::replace_last(s->meshesPath(), "/", "")
351  ) == 1;
352  hasParticles = std::count(
353  pList.paths->begin(),
354  pList.paths->end(),
355  auxiliary::replace_last(s->particlesPath(), "/", "")
356  ) == 1;
357  pList.paths->clear();
358  } else
359  {
360  hasMeshes = s->containsAttribute("meshesPath");
361  hasParticles = s->containsAttribute("particlesPath");
362  }
363 
364  if( hasMeshes )
365  {
366  Parameter< Operation::OPEN_PATH > pOpen;
367  pOpen.path = s->meshesPath();
368  IOHandler->enqueue(IOTask(&meshes, pOpen));
369 
370  meshes.readAttributes();
371 
372  /* obtain all non-scalar meshes */
373  IOHandler->enqueue(IOTask(&meshes, pList));
374  IOHandler->flush();
375 
376  Parameter< Operation::LIST_ATTS > aList;
377  for( auto const& mesh_name : *pList.paths )
378  {
379  Mesh& m = meshes[mesh_name];
380  pOpen.path = mesh_name;
381  aList.attributes->clear();
382  IOHandler->enqueue(IOTask(&m, pOpen));
383  IOHandler->enqueue(IOTask(&m, aList));
384  IOHandler->flush();
385 
386  auto att_begin = aList.attributes->begin();
387  auto att_end = aList.attributes->end();
388  auto value = std::find(att_begin, att_end, "value");
389  auto shape = std::find(att_begin, att_end, "shape");
390  if( value != att_end && shape != att_end )
391  {
392  MeshRecordComponent& mrc = m[MeshRecordComponent::SCALAR];
393  mrc.parent = m.parent;
394  IOHandler->enqueue(IOTask(&mrc, pOpen));
395  IOHandler->flush();
396  *mrc.m_isConstant = true;
397  }
398  m.read();
399  }
400 
401  /* obtain all scalar meshes */
402  Parameter< Operation::LIST_DATASETS > dList;
403  IOHandler->enqueue(IOTask(&meshes, dList));
404  IOHandler->flush();
405 
406  Parameter< Operation::OPEN_DATASET > dOpen;
407  for( auto const& mesh_name : *dList.datasets )
408  {
409  Mesh& m = meshes[mesh_name];
410  dOpen.name = mesh_name;
411  IOHandler->enqueue(IOTask(&m, dOpen));
412  IOHandler->flush();
413  MeshRecordComponent& mrc = m[MeshRecordComponent::SCALAR];
414  mrc.parent = m.parent;
415  IOHandler->enqueue(IOTask(&mrc, dOpen));
416  IOHandler->flush();
417  mrc.written() = false;
418  mrc.resetDataset(Dataset(*dOpen.dtype, *dOpen.extent));
419  mrc.written() = true;
420  m.read();
421  }
422  }
423 
424  if( hasParticles )
425  {
426  Parameter< Operation::OPEN_PATH > pOpen;
427  pOpen.path = s->particlesPath();
428  IOHandler->enqueue(IOTask(&particles, pOpen));
429 
430  particles.readAttributes();
431 
432  /* obtain all particle species */
433  pList.paths->clear();
434  IOHandler->enqueue(IOTask(&particles, pList));
435  IOHandler->flush();
436 
437  for( auto const& species_name : *pList.paths )
438  {
439  ParticleSpecies& p = particles[species_name];
440  pOpen.path = species_name;
441  IOHandler->enqueue(IOTask(&p, pOpen));
442  IOHandler->flush();
443  p.read();
444  }
445  }
446 
447  readAttributes();
448 }
