59  void
60  TracingJSON::invertShadow(
61  nlohmann::json & result,
62  nlohmann::json const & shadow )
63  {
64  if( !shadow.is_object() )
65  {
66  return;
67  }
68  std::vector< std::string > toRemove;
69  for( auto it = shadow.begin(); it != shadow.end(); ++it )
70  {
71  nlohmann::json & partialResult = result[ it.key() ];
72  if( partialResult.is_object() )
73  {
74  invertShadow( partialResult, it.value() );
75  if( partialResult.size() == 0 )
76  {
77  toRemove.emplace_back( it.key() );
78  }
79  }
80  else
81  {
82  toRemove.emplace_back( it.key() );
83  }
84  }
85  for( auto const & key : toRemove )
86  {
87  result.erase( key );
88  }
89  }
