77  Datatype
78  fromADIOS2Type( std::string const & dt )
79  {
80  static std::map< std::string, Datatype > map{
81  { "string", Datatype::STRING },
82  { "char", Datatype::CHAR },
83  { "signed char", Datatype::CHAR },
84  { "unsigned char", Datatype::UCHAR },
85  { "short", Datatype::SHORT },
86  { "unsigned short", Datatype::USHORT },
87  { "int", Datatype::INT },
88  { "unsigned int", Datatype::UINT },
89  { "long int", Datatype::LONG },
90  { "unsigned long int", Datatype::ULONG },
91  { "long long int", Datatype::LONGLONG },
92  { "unsigned long long int", Datatype::ULONGLONG },
93  { "float", Datatype::FLOAT },
94  { "double", Datatype::DOUBLE },
95  { "long double", Datatype::LONG_DOUBLE },
96  { "float complex", Datatype::CFLOAT },
97  { "double complex", Datatype::CDOUBLE },
98  { "long double complex", Datatype::CLONG_DOUBLE }, // does not exist as of 2.6.0 but might come later
99  { "uint8_t", Datatype::UCHAR },
100  { "int8_t", Datatype::CHAR },
101  { "uint16_t", determineDatatype< uint16_t >() },
102  { "int16_t", determineDatatype< int16_t >() },
103  { "uint32_t", determineDatatype< uint32_t >() },
104  { "int32_t", determineDatatype< int32_t >() },
105  { "uint64_t", determineDatatype< uint64_t >() },
106  { "int64_t", determineDatatype< int64_t >() }
107  };
108  auto it = map.find( dt );
109  if( it != map.end() )
110  {
111  return it->second;
112  }
113  else
114  {
115  std::cerr << "[ADIOS2] Warning: Encountered unknown ADIOS2 datatype,"
116  " defaulting to UNDEFINED." << std::endl;
117  return Datatype::UNDEFINED;
118  }
119  }

170  Datatype
171  attributeInfo(
172  adios2::IO & IO,
173  std::string const & attributeName,
174  bool verbose )
175  {
176  std::string type = IO.AttributeType( attributeName );
177  if( type.empty() )
178  {
179  if( verbose )
180  {
181  std::cerr << "[ADIOS2] Warning: Attribute with name "
182  << attributeName << " has no type in backend."
183  << std::endl;
184  }
185  return Datatype::UNDEFINED;
186  }
187  else
188  {
189  static AttributeInfo ai;
190  Datatype basicType = fromADIOS2Type( type );
191  auto size =
192  switchType< size_t >( basicType, ai, IO, attributeName );
193  Datatype openPmdType = size == 1
194  ? basicType
195  : size == 7 && basicType == Datatype::DOUBLE
196  ? Datatype::ARR_DBL_7
197  : toVectorType( basicType );
198  return openPmdType;
199  }
