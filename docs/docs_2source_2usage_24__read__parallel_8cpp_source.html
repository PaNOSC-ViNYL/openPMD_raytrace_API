33 int main(int argc, char *argv[])
34 {
35  MPI_Init(&argc, &argv);
36 
37  int mpi_size;
38  int mpi_rank;
39 
40  MPI_Comm_size(MPI_COMM_WORLD, &mpi_size);
41  MPI_Comm_rank(MPI_COMM_WORLD, &mpi_rank);
42 
43 
44  /* note: this scope is intentional to destruct the openPMD::Series object
45  * prior to MPI_Finalize();
46  */
47  {
48  Series series = Series(
49  "../samples/git-sample/data%T.h5",
50  Access::READ_ONLY,
51  MPI_COMM_WORLD
52  );
53  if( 0 == mpi_rank )
54  cout << "Read a series in parallel with " << mpi_size << " MPI ranks\n";
55 
56  MeshRecordComponent E_x = series.iterations[100].meshes["E"]["x"];
57 
58  Offset chunk_offset = {
59  static_cast< long unsigned int >(mpi_rank) + 1,
60  1,
61  1
62  };
63  Extent chunk_extent = {2, 2, 1};
64 
65  auto chunk_data = E_x.loadChunk<double>(chunk_offset, chunk_extent);
66 
67  if( 0 == mpi_rank )
68  cout << "Queued the loading of a single chunk per MPI rank from disk, "
69  "ready to execute\n";
70  series.flush();
71 
72  if( 0 == mpi_rank )
73  cout << "Chunks have been read from disk\n";
74 
75  for( int i = 0; i < mpi_size; ++i )
76  {
77  if( i == mpi_rank )
78  {
79  cout << "Rank " << mpi_rank << " - Read chunk contains:\n";
80  for( size_t row = 0; row < chunk_extent[0]; ++row )
81  {
82  for( size_t col = 0; col < chunk_extent[1]; ++col )
83  cout << "\t"
84  << '(' << row + chunk_offset[0] << '|' << col + chunk_offset[1] << '|' << 1 << ")\t"
85  << chunk_data.get()[row*chunk_extent[1]+col];
86  cout << std::endl;
87  }
88  }
89 
90  // this barrier is not necessary but structures the example output
91  MPI_Barrier(MPI_COMM_WORLD);
92  }
93  }
94 
95  // openPMD::Series MUST be destructed at this point
96  MPI_Finalize();
97 
98  return 0;
99 }
