196 void
197 Mesh::flush_impl(std::string const& name)
198 {
199  if(IOHandler->m_frontendAccess == Access::READ_ONLY )
200  {
201  for( auto& comp : *this )
202  comp.second.flush(comp.first);
203  } else
204  {
205  if( !written() )
206  {
207  if( scalar() )
208  {
209  MeshRecordComponent& mrc = at(RecordComponent::SCALAR);
210  mrc.m_writable->parent = parent;
211  mrc.parent = parent;
212  mrc.flush(name);
213  IOHandler->flush();
214  m_writable->abstractFilePosition = mrc.m_writable->abstractFilePosition;
215  mrc.abstractFilePosition = m_writable->abstractFilePosition.get();
216  abstractFilePosition = mrc.abstractFilePosition;
217  written() = true;
218  } else
219  {
220  Parameter< Operation::CREATE_PATH > pCreate;
221  pCreate.path = name;
222  IOHandler->enqueue(IOTask(this, pCreate));
223  for( auto& comp : *this )
224  comp.second.parent = this->m_writable.get();
225  }
226  }
227 
228  for( auto& comp : *this )
229  comp.second.flush(comp.first);
230 
231  flushAttributes();
232  }
233 }

235 void
236 Mesh::read()
237 {
238  using DT = Datatype;
239  Parameter< Operation::READ_ATT > aRead;
240 
241  aRead.name = "geometry";
242  IOHandler->enqueue(IOTask(this, aRead));
243  IOHandler->flush();
244  if( *aRead.dtype == DT::STRING )
245  {
246  std::string tmpGeometry = Attribute(*aRead.resource).get< std::string >();
247  if( "cartesian" == tmpGeometry )
248  setGeometry(Geometry::cartesian);
249  else if( "thetaMode" == tmpGeometry )
250  setGeometry(Geometry::thetaMode);
251  else if( "cylindrical" == tmpGeometry )
252  setGeometry(Geometry::cylindrical);
253  else if( "spherical" == tmpGeometry )
254  setGeometry(Geometry::spherical);
255  else
256  throw std::runtime_error("Unknown geometry " + tmpGeometry);
257  }
258  else
259  throw std::runtime_error("Unexpected Attribute datatype for 'geometry'");
260 
261  aRead.name = "dataOrder";
262  IOHandler->enqueue(IOTask(this, aRead));
263  IOHandler->flush();
264  if( *aRead.dtype == DT::CHAR )
265  setDataOrder(static_cast<DataOrder>(Attribute(*aRead.resource).get< char >()));
266  else if( *aRead.dtype == DT::STRING )
267  {
268  std::string tmpDataOrder = Attribute(*aRead.resource).get< std::string >();
269  if( tmpDataOrder.size() == 1 )
270  setDataOrder(static_cast<DataOrder>(tmpDataOrder[0]));
271  else
272  throw std::runtime_error("Unexpected Attribute value for 'dataOrder': " + tmpDataOrder);
273  }
274  else
275  throw std::runtime_error("Unexpected Attribute datatype for 'dataOrder'");
276 
277  aRead.name = "axisLabels";
278  IOHandler->enqueue(IOTask(this, aRead));
279  IOHandler->flush();
280  if( *aRead.dtype == DT::VEC_STRING )
281  setAxisLabels(Attribute(*aRead.resource).get< std::vector< std::string > >());
282  else if( *aRead.dtype == DT::STRING )
283  setAxisLabels({Attribute(*aRead.resource).get< std::string >()});
284  else
285  throw std::runtime_error("Unexpected Attribute datatype for 'axisLabels'");
286 
287  aRead.name = "gridSpacing";
288  IOHandler->enqueue(IOTask(this, aRead));
289  IOHandler->flush();
290  Attribute a = Attribute(*aRead.resource);
291  if( *aRead.dtype == DT::VEC_FLOAT )
292  setGridSpacing(a.get< std::vector< float > >());
293  else if( *aRead.dtype == DT::FLOAT )
294  setGridSpacing(std::vector< float >({a.get< float >()}));
295  else if( *aRead.dtype == DT::VEC_DOUBLE )
296  setGridSpacing(a.get< std::vector< double > >());
297  else if( *aRead.dtype == DT::DOUBLE )
298  setGridSpacing(std::vector< double >({a.get< double >()}));
299  else
300  throw std::runtime_error("Unexpected Attribute datatype for 'gridSpacing'");
301 
302  aRead.name = "gridGlobalOffset";
303  IOHandler->enqueue(IOTask(this, aRead));
304  IOHandler->flush();
305  if( *aRead.dtype == DT::VEC_DOUBLE )
306  setGridGlobalOffset(Attribute(*aRead.resource).get< std::vector< double > >());
307  else if( *aRead.dtype == DT::DOUBLE )
308  setGridGlobalOffset({Attribute(*aRead.resource).get< double >()});
309  else
310  throw std::runtime_error("Unexpected Attribute datatype for 'gridGlobalOffset'");
311 
312  aRead.name = "gridUnitSI";
313  IOHandler->enqueue(IOTask(this, aRead));
314  IOHandler->flush();
315  if( *aRead.dtype == DT::DOUBLE )
316  setGridUnitSI(Attribute(*aRead.resource).get< double >());
317  else
318  throw std::runtime_error("Unexpected Attribute datatype for 'gridUnitSI'");
319 
320  if( scalar() )
321  {
322  /* using operator[] will incorrectly update parent */
323  this->at(MeshRecordComponent::SCALAR).read();
324  } else
325  {
326  written() = false;
327  clear_unchecked();
328  written() = true;
329  Parameter< Operation::LIST_PATHS > pList;
330  IOHandler->enqueue(IOTask(this, pList));
331  IOHandler->flush();
332 
333  Parameter< Operation::OPEN_PATH > pOpen;
334  for( auto const& component : *pList.paths )
335  {
336  MeshRecordComponent& rc = (*this)[component];
337  pOpen.path = component;
338  IOHandler->enqueue(IOTask(&rc, pOpen));
339  *rc.m_isConstant = true;
340  rc.read();
341  }
342 
343  Parameter< Operation::LIST_DATASETS > dList;
344  IOHandler->enqueue(IOTask(this, dList));
345  IOHandler->flush();
346 
347  Parameter< Operation::OPEN_DATASET > dOpen;
348  for( auto const& component : *dList.datasets )
349  {
350  MeshRecordComponent& rc = (*this)[component];
351  dOpen.name = component;
352  IOHandler->enqueue(IOTask(&rc, dOpen));
353  IOHandler->flush();
354  rc.written() = false;
355  rc.resetDataset(Dataset(*dOpen.dtype, *dOpen.extent));
356  rc.written() = true;
357  rc.read();
358  }
359  }
360 
361  readBase();
362 
363  readAttributes();
364 }
