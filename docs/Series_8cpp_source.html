386 std::unique_ptr< Series::ParsedInput >
387 Series::parseInput(std::string filepath)
388 {
389  std::unique_ptr< Series::ParsedInput > input{new Series::ParsedInput};
390 
391 #ifdef _WIN32
392  if( auxiliary::contains(filepath, '/') )
393  {
394  std::cerr << "Filepaths on WINDOWS platforms may not contain slashes '/'! "
395  << "Replacing with backslashes '\\' unconditionally!" << std::endl;
396  filepath = auxiliary::replace_all(filepath, "/", "\\");
397  }
398 #else
399  if( auxiliary::contains(filepath, '\\') )
400  {
401  std::cerr << "Filepaths on UNIX platforms may not include backslashes '\\'! "
402  << "Replacing with slashes '/' unconditionally!" << std::endl;
403  filepath = auxiliary::replace_all(filepath, "\\", "/");
404  }
405 #endif
406  auto const pos = filepath.find_last_of(auxiliary::directory_separator);
407  if( std::string::npos == pos )
408  {
409  input->path = ".";
410  input->path.append(1, auxiliary::directory_separator);
411  input->name = filepath;
412  }
413  else
414  {
415  input->path = filepath.substr(0, pos + 1);
416  input->name = filepath.substr(pos + 1);
417  }
418 
419  input->format = determineFormat(input->name);
420 
421 #if defined(IS_GCC_48)
422  regex_t pattern;
423  if( regcomp(&pattern, "(.*)%(0[[:digit:]]+)?T(.*)", REG_EXTENDED) )
424  throw std::runtime_error(std::string("Regex for iterationFormat '(.*)%(0[[:digit:]]+)?T(.*)' can not be compiled!"));
425 
426  constexpr uint8_t regexGroups = 4u;
427  std::array< regmatch_t, regexGroups > regexMatch;
428  if( regexec(&pattern, input->name.c_str(), regexGroups, regexMatch.data(), 0) )
429  input->iterationEncoding = IterationEncoding::groupBased;
430  else
431  {
432  input->iterationEncoding = IterationEncoding::fileBased;
433 
434  if( regexMatch[0].rm_so == -1 || regexMatch[0].rm_eo == -1 )
435  throw std::runtime_error("Can not determine iterationFormat from filename " + input->name);
436 
437  if( regexMatch[1].rm_so == -1 || regexMatch[1].rm_eo == -1 )
438  throw std::runtime_error("Can not determine iterationFormat (prefix) from filename " + input->name);
439  else
440  input->filenamePrefix = std::string(&input->name[regexMatch[1].rm_so], regexMatch[1].rm_eo - regexMatch[1].rm_so);
441 
442  if( regexMatch[2].rm_so == -1 || regexMatch[2].rm_eo == -1 )
443  input->filenamePadding = 0;
444  else
445  input->filenamePadding = std::stoi(std::string(&input->name[regexMatch[2].rm_so], regexMatch[2].rm_eo - regexMatch[2].rm_so));
446 
447  if( regexMatch[3].rm_so == -1 || regexMatch[3].rm_eo == -1 )
448  throw std::runtime_error("Can not determine iterationFormat (postfix) from filename " + input->name);
449  else
450  input->filenamePostfix = std::string(&input->name[regexMatch[3].rm_so], regexMatch[3].rm_eo - regexMatch[3].rm_so);
451  }
452 
453  regfree(&pattern);
454 #else
455  std::regex pattern("(.*)%(0[[:digit:]]+)?T(.*)");
456  std::smatch regexMatch;
457  std::regex_match(input->name, regexMatch, pattern);
458  if( regexMatch.empty() )
459  input->iterationEncoding = IterationEncoding::groupBased;
460  else if( regexMatch.size() == 4 )
461  {
462  input->iterationEncoding = IterationEncoding::fileBased;
463  input->filenamePrefix = regexMatch[1].str();
464  std::string const& pad = regexMatch[2];
465  if( pad.empty() )
466  input->filenamePadding = 0;
467  else
468  {
469  if( pad.front() != '0' )
470  throw std::runtime_error("Invalid iterationEncoding " + input->name);
471  input->filenamePadding = std::stoi(pad);
472  }
473  input->filenamePostfix = regexMatch[3].str();
474  } else
475  throw std::runtime_error("Can not determine iterationFormat from filename " + input->name);
476 #endif
477 
478  input->filenamePostfix = cleanFilename(input->filenamePostfix, input->format);
479 
480  input->name = cleanFilename(input->name, input->format);
481 
482  return input;
483 }

485 void
486 Series::init(std::shared_ptr< AbstractIOHandler > ioHandler,
487  std::unique_ptr< Series::ParsedInput > input)
488 {
489  m_writable->IOHandler = ioHandler;
490  IOHandler = m_writable->IOHandler.get();
491  iterations.linkHierarchy(m_writable);
492 
493  m_name = std::make_shared< std::string >(input->name);
494 
495  m_format = std::make_shared< Format >(input->format);
496 
497  m_filenamePrefix = std::make_shared< std::string >(input->filenamePrefix);
498  m_filenamePostfix = std::make_shared< std::string >(input->filenamePostfix);
499  m_filenamePadding = std::make_shared< int >(input->filenamePadding);
500 
501  if(IOHandler->m_frontendAccess == Access::READ_ONLY || IOHandler->m_frontendAccess == Access::READ_WRITE )
502  {
503  /* Allow creation of values in Containers and setting of Attributes
504  * Would throw for Access::READ_ONLY */
505  auto oldType = IOHandler->m_frontendAccess;
506  auto newType = const_cast< Access* >(&m_writable->IOHandler->m_frontendAccess);
507  *newType = Access::READ_WRITE;
508 
509  if( input->iterationEncoding == IterationEncoding::fileBased )
510  readFileBased();
511  else
512  readGroupBased();
513 
514  if( iterations.empty() )
515  {
516  /* Access::READ_WRITE can be used to create a new Series
517  * allow setting attributes in that case */
518  written() = false;
519 
520  initDefaults();
521  setIterationEncoding(input->iterationEncoding);
522 
523  written() = true;
524  }
525 
526  *newType = oldType;
527  } else
528  {
529  initDefaults();
530  setIterationEncoding(input->iterationEncoding);
531  }
532 }

549 template< typename IterationsContainer >
550 void
551 Series::flushFileBased( IterationsContainer && iterationsToFlush )
552 {
553  if( iterationsToFlush.empty() )
554  throw std::runtime_error(
555  "fileBased output can not be written with no iterations." );
556 
557  if( IOHandler->m_frontendAccess == Access::READ_ONLY )
558  for( auto & i : iterationsToFlush )
559  {
560  if( *i.second.m_closed == Iteration::CloseStatus::ClosedInBackend )
561  {
562  // file corresponding with the iteration has previously been
563  // closed and fully flushed
564  // verify that there have been no further accesses
565  if( i.second.dirtyRecursive() )
566  {
567  throw std::runtime_error(
568  "[Series] Detected illegal access to iteration that "
569  "has been closed previously." );
570  }
571  continue;
572  }
573  i.second.flush();
574  if( *i.second.m_closed == Iteration::CloseStatus::ClosedInFrontend )
575  {
576  Parameter< Operation::CLOSE_FILE > fClose;
577  IOHandler->enqueue( IOTask( &i.second, std::move( fClose ) ) );
578  *i.second.m_closed = Iteration::CloseStatus::ClosedInBackend;
579  }
580  IOHandler->flush();
581  }
582  else
583  {
584  bool allDirty = dirty();
585  for( auto & i : iterationsToFlush )
586  {
587  if( *i.second.m_closed == Iteration::CloseStatus::ClosedInBackend )
588  {
589  // file corresponding with the iteration has previously been
590  // closed and fully flushed
591  // verify that there have been no further accesses
592  if (!i.second.written())
593  {
594  throw std::runtime_error(
595  "[Series] Closed iteration has not been written. This "
596  "is an internal error.");
597  }
598  if( i.second.dirtyRecursive() )
599  {
600  throw std::runtime_error(
601  "[Series] Detected illegal access to iteration that "
602  "has been closed previously." );
603  }
604  continue;
605  }
606  /* as there is only one series,
607  * emulate the file belonging to each iteration as not yet written
608  */
609  written() = false;
610  iterations.written() = false;
611 
612  std::stringstream iteration("");
613  iteration << std::setw(*m_filenamePadding) << std::setfill('0') << i.first;
614  std::string filename = *m_filenamePrefix + iteration.str() + *m_filenamePostfix;
615 
616  dirty() |= i.second.dirty();
617  i.second.flushFileBased(filename, i.first);
618 
619  iterations.flush(auxiliary::replace_first(basePath(), "%T/", ""));
620 
621  flushAttributes();
622 
623  if( *i.second.m_closed == Iteration::CloseStatus::ClosedInFrontend )
624  {
625  Parameter< Operation::CLOSE_FILE > fClose;
626  IOHandler->enqueue( IOTask( &i.second, std::move( fClose ) ) );
627  *i.second.m_closed = Iteration::CloseStatus::ClosedInBackend;
628  }
629 
630  IOHandler->flush();
631 
632  /* reset the dirty bit for every iteration (i.e. file)
633  * otherwise only the first iteration will have updates attributes */
634  dirty() = allDirty;
635  }
636  dirty() = false;
637  }
638 }

644 template< typename IterationsContainer >
645 void
646 Series::flushGroupBased( IterationsContainer && iterationsToFlush )
647 {
648  if( IOHandler->m_frontendAccess == Access::READ_ONLY )
649  for( auto & i : iterationsToFlush )
650  {
651  if( *i.second.m_closed == Iteration::CloseStatus::ClosedInBackend )
652  {
653  // file corresponding with the iteration has previously been
654  // closed and fully flushed
655  // verify that there have been no further accesses
656  if( i.second.dirtyRecursive() )
657  {
658  throw std::runtime_error(
659  "[Series] Illegal access to iteration " +
660  std::to_string( i.first ) +
661  " that has been closed previously." );
662  }
663  continue;
664  }
665  i.second.flush();
666  if( *i.second.m_closed == Iteration::CloseStatus::ClosedInFrontend )
667  {
668  // the iteration has no dedicated file in group-based mode
669  *i.second.m_closed = Iteration::CloseStatus::ClosedInBackend;
670  }
671  IOHandler->flush();
672  }
673  else
674  {
675  if( !written() )
676  {
677  Parameter< Operation::CREATE_FILE > fCreate;
678  fCreate.name = *m_name;
679  IOHandler->enqueue(IOTask(this, fCreate));
680  }
681 
682  iterations.flush( auxiliary::replace_first( basePath(), "%T/", "" ) );
683 
684  for( auto & i : iterationsToFlush )
685  {
686  if( *i.second.m_closed == Iteration::CloseStatus::ClosedInBackend )
687  {
688  // file corresponding with the iteration has previously been
689  // closed and fully flushed
690  // verify that there have been no further accesses
691  if (!i.second.written())
692  {
693  throw std::runtime_error(
694  "[Series] Closed iteration has not been written. This "
695  "is an internal error.");
696  }
697  if( i.second.dirtyRecursive() )
698  {
699  throw std::runtime_error(
700  "[Series] Illegal access to iteration " +
701  std::to_string( i.first ) +
702  " that has been closed previously." );
703  }
704  continue;
705  }
706  if( !i.second.written() )
707  {
708  i.second.m_writable->parent = getWritable(&iterations);
709  i.second.parent = getWritable(&iterations);
710  }
711  i.second.flushGroupBased(i.first);
712  if( *i.second.m_closed == Iteration::CloseStatus::ClosedInFrontend )
713  {
714  // the iteration has no dedicated file in group-based mode
715  *i.second.m_closed = Iteration::CloseStatus::ClosedInBackend;
716  }
717  }
718 
719  flushAttributes();
720  IOHandler->flush();
721  }
722 }

1008  std::function<std::tuple<bool, int>(std::string const &)>
1009  buildMatcher(std::string const &regexPattern) {
1010 #if defined(IS_GCC_48)
1011  auto pattern = std::shared_ptr< regex_t >(new regex_t, [](regex_t* p){ regfree(p); delete p; });
1012  if( regcomp(pattern.get(), regexPattern.c_str(), REG_EXTENDED) )
1013  throw std::runtime_error(std::string("Regex for name '") + regexPattern + std::string("' can not be compiled!"));
1014 
1015  return [pattern](std::string const& filename) -> std::tuple< bool, int >
1016  {
1017  std::array< regmatch_t, 2 > regexMatches;
1018  bool match = !regexec(pattern.get(), filename.c_str(), regexMatches.size(), regexMatches.data(), 0);
1019  int padding = match ? regexMatches[1].rm_eo - regexMatches[1].rm_so : 0;
1020  return std::tuple< bool, int >{match, padding};
1021  };
1022 #else
1023  std::regex pattern(regexPattern);
1024 
1025  return [pattern](std::string const &filename) -> std::tuple<bool, int> {
1026  std::smatch regexMatches;
1027  bool match = std::regex_match(filename, regexMatches, pattern);
1028  int padding = match ? regexMatches[1].length() : 0;
1029  return std::tuple<bool, int>{match, padding};
